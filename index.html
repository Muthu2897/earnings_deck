Code Sample — KPI Trend Calculation


import pandas as pd

def growth_rate(current, prior):
    # Simple QoQ growth rate calculation
    return (current / prior - 1.0) * 100.0

revenue_q2 = 3_200_000_000
revenue_q1 = 3_005_000_000
print(f"QoQ revenue growth: {growth_rate(revenue_q2, revenue_q1):.2f}%")
