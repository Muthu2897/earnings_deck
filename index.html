ETL Validation — Sample Task


import pandas as pd

# Load anonymized transactions and compute net interest income (NII)
df = pd.read_csv("transactions.csv")

nii = (
    df.assign(
        interest_income=lambda d: d["loan_bal"] * d["rate_annual"] / 12,
        interest_expense=lambda d: d["deposit_bal"] * d["cost_of_funds"] / 12,
    )["interest_income"].sum()
    - df["interest_expense"].sum()
)

print(f"Net Interest Income (monthly): {nii:,.2f}")
